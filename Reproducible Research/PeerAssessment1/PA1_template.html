totalNA <- sum(is.na(data))

stepsList <- split(data$steps, factor(data$interval))
stepsList <- lapply(stepsList, 
        function(stepsList){
                stepsList[which(is.na(stepsList))] <- mean(stepsList, na.rm = TRUE)
                stepsList
        })

fillData <- data
